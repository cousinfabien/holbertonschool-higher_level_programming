# Cas 2 : liste déjà triée
>>> ml = MyList([1, 2, 3])
>>> ml.print_sorted()
[1, 2, 3]

# Cas 3 : nombres négatifs
>>> ml = MyList([0, -1, -5, 3])
>>> ml.print_sorted()
[-5, -1, 0, 3]

# Cas 4 : doublons
>>> ml = MyList([2, 2, 1, 3])
>>> ml.print_sorted()
[1, 2, 2, 3]

# Cas 5 : liste vide
>>> ml = MyList()
>>> ml.print_sorted()
[]

# Cas 6 : vérification que la liste originale n'est pas modifiée
>>> ml = MyList([3, 1, 2])
>>> ml.print_sorted()
[1, 2, 3]
>>> ml
[3, 1, 2]

# Cas 7 : liste avec un seul élément
>>> ml = MyList([42])
>>> ml.print_sorted()
[42]

# Cas 8 : grands nombres
>>> ml = MyList([999999999, -999999999])
>>> ml.print_sorted()
